CREATE TABLE `ad_cost` (
  `id` int unsigned NOT NULL AUTO_INCREMENT,
  `date` date NOT NULL COMMENT '时间',
  `project_id` int unsigned DEFAULT '0' COMMENT '游戏ID',
  `appid` int unsigned DEFAULT '0' COMMENT '游戏ID',
  `os` tinyint unsigned DEFAULT '0' COMMENT '系统',
  `country_code` varchar(64) DEFAULT '' COMMENT '国家编码',
  `partner_id` int unsigned DEFAULT '0' COMMENT '广告渠道ID',
  `ad_directorid` int unsigned DEFAULT '0' COMMENT '广告负责人ID',
  `campaignid` varchar(150) NOT NULL DEFAULT '' COMMENT '渠道campaignID',
  `adgroupid` varchar(150) NOT NULL DEFAULT '' COMMENT '渠道广告ID',
  `creativeid` varchar(150) NOT NULL DEFAULT '' COMMENT '渠道创意ID',
  `cost` decimal(10,2) unsigned NOT NULL DEFAULT '0.00' COMMENT '花费金额',
  `charge` decimal(10,2) unsigned NOT NULL DEFAULT '0.00' COMMENT '付费金额',
  `install` int unsigned DEFAULT '0' COMMENT '安装数',
  `convert` bigint unsigned DEFAULT '0' COMMENT '转化数',
  `click` bigint unsigned DEFAULT '0' COMMENT '点击',
  `exposure` bigint unsigned DEFAULT '0' COMMENT '曝光',
  `origin_cost` decimal(10,2) NOT NULL DEFAULT '0.00' COMMENT '返点后的消耗',
  `rebate` decimal(10,2) NOT NULL DEFAULT '0.00' COMMENT '返点比例',
  `account_id` varchar(50) DEFAULT '',
  PRIMARY KEY (`id`,`date`),
  UNIQUE KEY `unique_login_key` (`date`,`project_id`,`account_id`,`appid`,`os`,`country_code`,`partner_id`,`ad_directorid`,`campaignid`(50),`adgroupid`(50),`creativeid`(50))
) ENGINE=InnoDB AUTO_INCREMENT=56 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
