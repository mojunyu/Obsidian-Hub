package third  
  
import (  
    "context"

type WechatOauth struct {

}  
  
type WechatRequestAuthInfo struct {  
    JsCode   string `json:"js_code"`

type WechatSession struct {  
    Openid     string `json:"openid"`  
    SessionKey string `json:"session_key"`  
    Unionid    string `json:"unionid"`  
    ErrCode    int    `json:"errcode"`

}  
  
// WechatThirdInfo 存在 third_0.third_info 的 JSON（来自 Code2Session），供支付等使用  
type WechatThirdInfo struct {  
    Openid     string `json:"openid"`

type WechatStableAccessTokenResp struct {

    ForceRefresh bool   `json:"force_refresh,omitempty"`  
}  
  
var (  
    wxApiUrl  = "https://api.weixin.qq.com"  
    wxUnivUrl = "https://univapi.wechat.com"  
)  
  
func (w *WechatOauth) Authorization(authInfo string, code string, extra string) (thirdAuthInfo OAuthUserThird, bizErr ecode.BizError) {  
    var (  
       authUserInfo WechatRequestAuthInfo  
       session      WechatSession  
    )  
  
    if authInfo == "" {  
       return thirdAuthInfo, ecode.AUTH_ERROR  
    }  
  
    if err := json.Unmarshal([]byte(authInfo), &authUserInfo); err != nil {  
       logs.Error("json unmarshal wechat auth info err: %v", err)  
    }  
  
    if authUserInfo.JsCode == "" {  
       return thirdAuthInfo, ecode.AUTH_ERROR  
    }  
  
    // 2.2 code2Session 登录凭证校验获取 openid 和 unionid    session, bizErr = w.Code2Session(authUserInfo.Platform, authUserInfo.JsCode)    if bizErr.Code() != 0 {  
       return thirdAuthInfo, bizErr  
    }  
  
    // 将 Code2Session 的 openid、session_key、unionid 写入 third_info（JSON），供后续微信支付等场景使用  
    thirdInfo, _ := json.Marshal(WechatThirdInfo{  
       Openid:     session.Openid,  
       SessionKey: session.SessionKey,  
       Unionid:    session.Unionid,  
       Platform:   authUserInfo.Platform,  
    })  
  
    return OAuthUserThird{  
       OpenId:    session.Openid,  
       AppId:     w.AppConfig.AppId,  
       RegType:   4, // RegTypeWeChat  
       Avatar:    "",  
       Nickname:  "",  
       Gender:    0,  
       UnionId:   session.Unionid,  
       ThirdInfo: string(thirdInfo),  
       Prefix:    "wx_",  
    }, ecode.BizError{}  
}  
  
// Code2Session 获取用户信息：2.2 code2Session 登录凭证校验  
func (w *WechatOauth) Code2Session(platform, code string) (session WechatSession, bizError ecode.BizError) {  
    sessionUrl := ""  
    if platform == "devtools" {  
       sessionUrl = wxApiUrl + "/sns/jscode2session"  
    } else {  
       sessionUrl = wxUnivUrl + "/sns/jscode2session"  
    }  
    requestMap := map[string]string{  
       "appid":      w.AppConfig.AppId,  
       "secret":     w.AppConfig.AppKey,  
       "js_code":    code,  
       "grant_type": "authorization_code",  
    }  
    w.Http.Cli.Timeout = time.Second * 10  
    w.Http.Params = requestMap  
    resp, err := w.Http.Request(context.Background(), sessionUrl, "GET", nil)  
    if err != nil {  
       logs.Error("wechat code2session request err: %v", err)  
       return session, ecode.AUTH_ERROR  
    }  
    defer resp.Body.Close()  
  
    if err = json.NewDecoder(resp.Body).Decode(&session); err != nil {  
       logs.Error("json decode wechat code2session resp err: %v", err)  
       return session, ecode.AUTH_ERROR  
    }  
    // 微信的错误码判断  
    if session.ErrCode != 0 {  
       logs.Error("wechat code2session errcode: %d, errmsg: %s", session.ErrCode, session.ErrMsg)  
       return session, ecode.AUTH_ERROR  
    }  
  
    return session, ecode.BizError{}  
}  
  
// GetStableAccessToken 获取 access_token (部分业务可能需要)  
func (w *WechatOauth) GetStableAccessToken() (token WechatStableAccessTokenResp, err error) {  
    defer func() { w.Http.Payload = nil }()  
  
    tokenURL := wxApiUrl + "/cgi-bin/stable_token"  
    reqBody := WechatStableAccessTokenReq{  
       GrantType:    "client_credential",  
       Appid:        w.AppConfig.AppId,  
       Secret:       w.AppConfig.AppKey,  
       ForceRefresh: false,  
    }  
    bodyBytes, err := json.Marshal(reqBody)  
    if err != nil {  
       return  
    }  
  
    w.Http.Cli.Timeout = time.Second * 10  
    w.Http.Params = nil  
    w.Http.Payload = bodyBytes  
  
    resp, err := w.Http.Request(context.Background(), tokenURL, "POST", nil)  
    if err != nil {  
       return  
    }  
    defer resp.Body.Close()  
  
    if err = json.NewDecoder(resp.Body).Decode(&token); err != nil {  
       return  
    }  
    if token.ErrCode != 0 {  
       return token, fmt.Errorf("errcode: %d, errmsg: %s", token.ErrCode, token.ErrMsg)  
    }  
  
    return token, nil  
}  
  
func NewWechatOauth(config model.ThirdLoginType) (OauthBase, ecode.BizError) {  
    if config.AppId == "" || config.AppKey == "" {  
       return &WechatOauth{}, ecode.NOT_CONFIG_WECHAT_APP_INFO  
    }  
    return &WechatOauth{  
       Oauth{  
          Url:       wxUnivUrl,  
          AppConfig: config,  
          Http:      utils.NewHttpClient(),  
       },  
    }, ecode.BizError{}  
}  
  
// CalcPaySig pay_sig 签名算法（支付请求签名）  
// uri: 当前请求的支付 API 的 uri 部分，不带 query_string，例如：/wxa/game/getbalance、/wxa/game/pay  
// postBody: http POST 的数据包体  
// appkey: 对应环境的 AppKeyfunc CalcPaySig(uri, postBody, appkey string) string {  
    needSignMsg := uri + "&" + postBody  
    h := hmac.New(sha256.New, []byte(appkey))  
    h.Write([]byte(needSignMsg))  
    return hex.EncodeToString(h.Sum(nil))  
}  
  
// CalcSignature 用户登录态 signature 签名算法  
// postBody: http POST 的数据包体  
// sessionKey: 当前用户有效的 session_key，参考 auth.code2Session 接口  
func CalcSignature(postBody, sessionKey string) string {  
    h := hmac.New(sha256.New, []byte(sessionKey))  
    h.Write([]byte(postBody))  
    return hex.EncodeToString(h.Sum(nil))  
}

// GetGameBalance 查询游戏币余额（pay_v2.getBalance）  
// 文档：https://gamemp.weixin.qq.com/doc/overseas-game/guide/pay.html#%E6%9F%A5%E8%AF%A2%E6%B8%B8%E6%88%8F%E5%B8%81%E4%BD%99%E9%A2%9D%EF%BC%88pay-v2-getBalance%EF%BC%89  
func (w *WechatOauth) GetGameBalance(platform, openid, sessionKey, zoneID string, env int, userIP, payAppKey string, ts int64) (data WechatGameBalanceResp, bizErr ecode.BizError) {  
    if openid == "" || sessionKey == "" || zoneID == "" {  
       return data, ecode.INVALID_PARAM  
    }  
    if ts <= 0 {  
       ts = time.Now().Unix()  
    }  
    if payAppKey == "" {  
       // 兜底：若业务侧未传入支付 AppKey，则使用当前 third_login.wechat 的 AppKey       payAppKey = w.AppConfig.AppKey    }  
    if payAppKey == "" {  
       return data, ecode.INVALID_PARAM  
    }  
  
    uri := "/wxa/game/getbalance"  
    reqBody := WechatGameBalanceReq{  
       Openid: openid,  
       Ts:     ts,  
       ZoneID: zoneID,  
       Env:    env,  
       UserIP: userIP,  
    }  
    bodyBytes, err := json.Marshal(reqBody)  
    if err != nil {  
       logs.Error("wechat getbalance json marshal req err: %v", err)  
       return data, ecode.SYSTEM_BUSY  
    }  
    postBody := string(bodyBytes)  
  
    // 构造签名  
    signature := CalcSignature(postBody, sessionKey)  
    paySig := CalcPaySig(uri, postBody, payAppKey)  
  
    // 注意apiURL和参数拼装  
    apiBase := wxUnivUrl  
    if platform == "devtools" {  
       apiBase = wxApiUrl  
    }  
    apiURL := apiBase + uri  
  
    // 获取 access_token    accessToken, err := w.GetStableAccessToken()  
    if err != nil || accessToken.AccessToken == "" {  
       if err != nil {  
          logs.Error("wechat getbalance get access_token err: %v", err)  
       } else {  
          logs.Error("wechat getbalance get access_token empty")  
       }  
       return data, ecode.SYSTEM_BUSY  
    }  
  
    params := fmt.Sprintf(  
       "?access_token=%s&signature=%s&sig_method=hmac_sha256&pay_sig=%s",  
       accessToken.AccessToken, signature, paySig,  
    )  
    fullURL := apiURL + params  
  
    // 配置http请求  
    w.Http.Cli.Timeout = time.Second * 10  
    w.Http.Params = nil // 已经拼在URL里  
    w.Http.Payload = bodyBytes  
  
    resp, err := w.Http.Request(context.Background(), fullURL, "POST", nil)  
    if err != nil {  
       logs.Error("wechat getbalance request err: %v", err)  
       return data, ecode.SYSTEM_BUSY  
    }  
    defer resp.Body.Close()  
  
    if err = json.NewDecoder(resp.Body).Decode(&data); err != nil {  
       logs.Error("json decode wechat getbalance resp err: %v", err)  
       return data, ecode.SYSTEM_BUSY  
    }  
    if data.ErrCode != 0 {  
       logs.Error("wechat getbalance errcode: %d, errmsg: %s", data.ErrCode, data.ErrMsg)  
       switch data.ErrCode {  
       case -1:  
          return data, ecode.SYSTEM_BUSY  
       case 90010:  
          return data, ecode.LOGIN_EXPIRED  
       case 90011:  
          return data, ecode.AUTH_ERROR  
       case 90018:  
          return data, ecode.INVALID_PARAM  
       default:  
          return data, ecode.AUTH_ERROR  
       }  
    }  
    return data, ecode.BizError{}  
}  
  
// CheckSessionKey 校验服务器保存的 session_key 是否有效。  
// signature = hmac_sha256(session_key, "")  
// 文档（海外版同形态）：https://univapi.wechat.com/wxa/checksession  
func (w *WechatOauth) CheckSessionKey(platform, openid, sessionKey string) (ok bool, bizErr ecode.BizError) {  
    if openid == "" || sessionKey == "" {  
       return false, ecode.INVALID_PARAM  
    }  
  
    accessToken, err := w.GetStableAccessToken()  
    if err != nil || accessToken.AccessToken == "" {  
       if err != nil {  
          logs.Error("wechat checksession get access_token err: %v", err)  
       } else {  
          logs.Error("wechat checksession get access_token empty")  
       }  
       return false, ecode.SYSTEM_BUSY  
    }  
  
    apiBase := wxUnivUrl  
    if platform == "devtools" {  
       apiBase = wxApiUrl  
    }  
    checkUrl := apiBase + "/wxa/checksession"  
  
    // 构造签名：hmac_sha256(session_key, "")  
    signature := CalcSignature("", sessionKey)  
  
    req := map[string]string{  
       "access_token": accessToken.AccessToken,  
       "openid":       openid,  
       "signature":    signature,  
       "sig_method":   "hmac_sha256",  
    }  
  
    var respBody WechatCheckSessionResp  
    w.Http.Cli.Timeout = time.Second * 10  
    w.Http.Params = req  
    w.Http.Payload = nil // GET 请求不应携带 body    resp, err := w.Http.Request(context.Background(), checkUrl, "GET", nil)  
    if err != nil {  
       logs.Error("wechat checksession request err: %v", err)  
       return false, ecode.SYSTEM_BUSY  
    }  
    defer resp.Body.Close()  
  
    if err = json.NewDecoder(resp.Body).Decode(&respBody); err != nil {  
       logs.Error("json decode wechat checksession resp err: %v", err)  
       return false, ecode.SYSTEM_BUSY  
    }  
    if respBody.ErrCode != 0 {  
       logs.Error("wechat checksession errcode: %d, errmsg: %s", respBody.ErrCode, respBody.ErrMsg)  
       switch respBody.ErrCode {  
       case -1:  
          return false, ecode.SYSTEM_BUSY  
       case 90010:  
          return false, ecode.LOGIN_EXPIRED  
       case 90011:  
          return false, ecode.AUTH_ERROR  
       case 90018:  
          return false, ecode.INVALID_PARAM  
       default:  
          return false, ecode.AUTH_ERROR  
       }  
    }  
    return true, ecode.BizError{}  
}
